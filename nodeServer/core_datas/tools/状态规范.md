main_virtual.status是记载main_virtual内核状态的文件

其内部维护一个JSON文本表示的对象

其内部应该是这样的结构

```json
{
    system:{
        main_prompt:string,//静态的
        character_reference:string,//静态的,因为现版本是插入主提示词里面的，所以一般为空字符串
        events:string,//因为event基本不更新，所以当前版本锁为一个string就行，之后改
        workspace:[
            {
                index:string,//以ws开头，然后是序号
                current:string//这里写着%[文件的完整路径]或者直接是文件内容
            }
        ],
        //没有object_network
        pulled_info:[
            {
                index:string,//以pi开头，然后是数字
                current:string//完整的文件路径，不加%[]
            }
        ],
        step_progres:[
        	{
                index:string,//以sp开头，然后是数字
                status:string,//可以是pending,processing,completed其中之一
                current:string//详细的计划的内容
            }
        ],
    }
    context:Message[]//这个就直接是一个Message数组了,虽然是google格式输入的，但是被接受会转换成deepseek的Message:TextMessage[]
}
```
